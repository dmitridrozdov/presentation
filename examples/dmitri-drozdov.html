            def big_commerce_request_demo(title, driver):
              try:

                  click_web_element_by_class_name_and_name(driver,

                  click_web_element_by_css_selector_and_name(driver, 
                  'a[role=button]', 'REQUEST A DEMO')

                  input_ids = ['FirstName', 'LastName', 
                  'Email', 'Company', 
                    'Projected_Annual_Revenue__c', 'Phone', 
                    'Country']

                  input_values = ['Dmytro', 'Drozdov', 
                  'dm.drozdov@gmail.com', 
                    'BigCommerce', "I'm not sure", 
                    '041234567', 'Australia']

                  [input_text_by_id(driver, el_id, value) 
                    for el_id, value in zip(input_ids, input_values)]

                  return TestResult(title, 'Pass', 
                          'Requested Demo from BigCommerce web site')

              except Exception as e:
                  return TestResult(title, 'Fail', str(e))
